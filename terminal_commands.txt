                                                                      GIT
--1. Проверить информацию о юзере
git config --list --global
--2. Создать папку удаленного репозиторя на локаной машине
git clone <http>
--3.Отследить изменения
git status
--4.Добавить в трекинг файл
git add
--5.Подтвердить изменения
git commit -m "text"
--6.Отправить файл в удаленный репозиторий
git push
--7.Проверить содержимое файла
cat
--8.Показать изменения в удаленном репозитории
git fetch
--9.Изменить мастер ветку на локальной машине
git pull
--10.Показать логи
git log
--11.Показать логи автора
git log --author <user>
--12. Вывести информацию о коммите
git show <hash>
--13.Показать последний коммит
git show
--14.Посмореть изменения автора
git blame 
--15.Узнать кто вносил изменения строки
git blame <file> | <line>
--16.Узнать что вносил юзер
git blame <file> | grep <user>
--17.Вывести информацию об изменении в файле
git diff
--18. Добавить строку в файл
echo "text">><file>
--19.Вывести файлы которые находяться в репозитории
ls
--20.Создать новый файл
echo "new file" >> <file>
--21.Коммит всех файлов
git commit -a -m
--22.Откатить на один коммит назад
git reset HEAD~1
--23.Отменить попытку слияния
git merge  --abort
--24. Отменить изменения в файле
git checkout
--25. Отменить изменения во всех файлах
git checkout .
--26. Временное хранилище для изменений
git stash
--27.Показать новые строки
git stash pop
--28.Удалить информацию из хранилища
git stash clear
--29.Создать новую ветку
git branch <name>
--30. Показать ветки 
git brach
--31.Переключиться на ветку
git checkout <name> 
--32.Синхронизировать на локальном и удаленном репозитории ветки
git push -u origin <name>
--33.Залить изменения в основную ветку
git merge <name>
                                                                        BASH
--1.Показать имя папки
pwd
--2.Создать папку
mkdir <name>
--3.Перейти в папку 
cd
--4.Очистить терминал от команд
clear
--5.Показать файлы в директории
ls
--6.Показать скрытые файлы
ls-a
--7.Показать содержимое
ls -R
--8.Показать ключи нужной команды
man ls
--9.Удалить пустую папку
rmdir <name>
--10.Удалить файл
rm <name>
--11.Удалить не пустую папку
rm -R <name>
--12. Переместить файл в другую папку
mv <name/file> <name/file>
--13.Копировать файл в папку
cp <name/file> <name>
--14.Переместить все файлы из папки в другую папку
ls <name/*> <name>
--15.Открыть файл в отдельном окне
less <file>
--16.Открыть текстовый редактор 
nano <file>
--17.Открыть редактор
vi <file>
--18. Найти текст в файле
grep <text> <file>
--19.Найти файлы по имени
find . -name <file>
--20. Пинговать сайт
ping <site>
--21. Отправить n пакетов 
ping -c <n> <site>
